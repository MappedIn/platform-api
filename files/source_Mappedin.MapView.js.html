/**
 * The base MapView. This will either be 2D or 3D. Make sure you run one of Mappedin.{{#crossLink "Mappedin/test3D:method"}}{{/crossLink}}, .{{#crossLink "Mappedin/forceTest:method"}}{{/crossLink}}, .{{#crossLink "Mappedin/force2D:method"}}{{/crossLink}} or .{{#crossLink "Mappedin/force3D:method"}}{{/crossLink}} before you try an instantiate one.
 *
 * @class MapView
 * @constructor
 * @param container {Div} The div to place the MapView in.

 * @param options {Object} Some options the MapView can use.
 	@param [options.antialias=true] {Boolean} Controls whether antialiasing is on. Defaults to true, as long as devicePixelRatio is 1. This is very expensive on high resolution screens.
 	@param [options.onDataLoaded] {Function} A callback executed with the 3D files have been downloaded for the first map, and are starting to load into memory. The MapView is fully functional at this point, but some things will still be popping in.
 	@param [options.onFirstMapLoaded] {Function} A callback executed when the first Map is fully loaded. This means onDataLoaded has already been fired (if specified) and then all the textures and other resources have finished popping in.
	@param [options.firstMap] {MappedinMap} If you don't want to load the default Map first, set the Map you want to load first here. Useful for deep linking to a certain floor/location. Since you won't have the Map object/ID until you get the {{#crossLink "MappedinVenue"}}{{/crossLink}} data, there is a callback you can set on Mappedin.{{#crossLink "Mappedin/initialize:method"}}{{/crossLink}}to fill this in.

	/**
	* Calculates the length of the connected path in list of Nodes. Breaks in the path (to go to other floors) count as 0 distance. The Z value is also ignored.
	*
	* @method getPathLength
	* @param path {[MappedinNode]} An array of Nodes on a path, probably from a Node's {{#crossLink "MappedinNode/directionsTo:method"}}{{/crossLink}} call.
	* @return {Integer} The 2D length of the path
	*/
	this.getPathLength = function(path) {
		var length = 0;
		var map = path[0].map;
		for (var i = 0, iLen = path.length - 1; i < iLen; ++i) {
			var node = path[i];
			var nextNode = path[i + 1];
			if (nextNode.map == map) {
				length += Math.sqrt(Math.pow((nextNode.x - node.x), 2) + Math.pow((nextNode.y - node.y), 2));
			} else {
				map = nextNode.map;
			}
		}
		return length;
	};

	/**
	* Finds the main Location associated with a Polygon. This means a Location
	* attached to the Polygon that has no parents, or, if there are none of those,
	* a Location nearest the top of some hierarchy that does have the Polygon attached.
	*
	* This means if there are multiple hierarchies of Locations attached to the Polygon,
	* the one that gets returned is not guaranteed to be what you want.
	*
	* @method getPrimaryLocationForPolygon
	* @param polygon {MappedinPolygon} The Polygon you want the primary Location of.
	* @return {MappedinLocation}
	*/
	this.getPrimaryLocationForPolygon = function(polygon) {
		// Handle polygon ids
		if (typeof polygon == "string") {
			if (scope.venue != undefined && Array.isArray(scope.venue.polygons)) {
				polygon = scope.venue.polygons.find(function (p) {
					return p.id == polygon;
				});
			} else {
				return null;
			}
		}
		var location = null;
		if (polygon != null && polygon.locations != undefined && polygon.locations.length > 0) {
			// If the polygon has locations, return one of them no matter what
			location = polygon.locations[0];
			for (var i = 0, iLen = polygon.locations.length; i < iLen; i++) {
				var testLoc = polygon.locations[i];
				// Check for weird hierarchies, like location A->B->C where B and C have
				// polygon X, but the parent A doesn't
				while (testLoc.parent != undefined) {
					testLoc = testLoc.parent;
					if (testLoc.polygons.includes(polygon)) {
						location = testLoc;
					}
				}
				// If we ever have a Location with no parent attached to the polygon, it's the primary.
				// If there are more than one, which one will be returned is not defined.
				if (testLoc.parent == undefined && testLoc.polygons.includes(polygon)) {
					return testLoc;
				}
			}
		}
		// Return our best guess at "most primary"
		return location;
	};
