import p2 from 'p2';
import PubSub from './Mappedin.PubSub.js';

var stopPropagation = function(event) {
	event.stopPropagation();
};

/**
 * A Marker is a div with whatever HTML you provide, anchored it to a `position` in 3D space. Make your own with {{#crossLink "MapView/createMarker:method"}}{{/crossLink}}.
 *
 * @class Marker
 */
var Marker = function (mapView, options) {
	Object.assign(this, PubSub.prototype);
	var marker = this;
	var div = document.createElement('div');
	marker.div = div;
	marker._mMapId = options.map ? options.map.id || options.map : mapView.currentMap;
	marker._mMarkerBuffer = options.markerBuffer || 2;

	marker._mLockToAnchor = options.lockToAnchor == true;
	marker._mIgnoreCollisions = options.ignoreCollisions == true;
	//marker._mKinematic = options.kinematic == true
	marker._mRotateWithCamera = options.rotateWithCamera == true;
	marker._mIntitialRotation = options.rotation ? options.rotation : 0;
	marker.m_translateX = 0;
	marker.m_translateY = 0;
	marker.m_rotateZ = 0;

	// Set up inner div
	div.className = options.className || options.class;// or default marker class
	div.innerHTML = options.html;

	if (options.position) {
		div.style.position = 'absolute';
		div.style.top = "0px";
		div.style.left = "0px";
		div.style.willChange = 'transform';
	}

	div.addEventListener("mousedown", stopPropagation); // Markers can never click through to the mapview
	marker._oldPointerEvents = div.style.pointerEvents;

	marker._mPosition = options.position;
	marker._mOriginalPosition = options.position;
	marker._mProjectedPosition = [0, 0];

	this.destroy = function () {
		this.publish('destroy');
		div.removeEventListener("mousedown", stopPropagation);
		// Do any cleanup
		for (var eventName in this.subscribers) {
			var eventSubscribers = this.subscribers[eventName];
			var numFns = eventSubscribers.length;
			for (var iter = numFns - 1; iter >= 0; --iter) {
				marker.off(eventName, eventSubscribers[iter]);
			}
		}
	};

	// Check if a marker is currently colliding with something
	this.isColliding = function() {
		return !marker._mIgnoreCollisions && marker._mShadowElement._mNumCollisions > 0;
	};

	// Makes a marker visible after camera movement
	this.show = function() {
		if (marker._mHidden == true) {
			marker._mHidden = false;
			if (marker._mAnchor) {
				p2.vec2.copy(marker._mShadowElement.position, marker._mAnchor.position);
			}

			div.style.opacity = marker._oldOpacity;
			div.style.pointerEvents = marker._oldPointerEvents;
		}
	};

	// Hides a marker in preparation for camera movement
	this.hide = function () {
		if (marker._mHidden != true) {
			marker._mHidden = true;
			marker._oldOpacity = div.style.opacity;
			div.style.opacity = 0;
			marker._oldPointerEvents = div.style.pointerEvents;
			div.style.pointerEvents = "none";
		}
	};

	this.setPosition = function (position) {
		if (position && position.x && position.y) {
			marker._mPosition = position;
			mapView.tryRendering();
		}
	};

	this.setTransform = function(transforms) {
		marker.m_translateX = transforms.x;
		marker.m_translateY = transforms.y;
		marker.m_rotateZ = transforms.rotateZ;

		var translate = 'translate(' + transforms.x + 'px, ' + transforms.y + 'px)';
		var rotate = 'rotate(' + transforms.rotateZ + 'rad)';
		var transform = translate + ' ' + rotate;

		marker.div.style.transform = transform;
		this.publish('update-position', { x: transforms.x, y: transforms.y, rotateZ: transforms.rotateZ });
	};

	this.addEventListener = function(ev, response) {
		marker.div.addEventListener(ev, response);
	};
};

export default Marker;
